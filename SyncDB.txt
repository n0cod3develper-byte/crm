"""
===========================================================

Compatible con Windows Server 2008 R2

    - Lee datos desde MySQL
    - Inserta/Actualiza en PostgreSQL
    - Evita duplicados

===========================================================
"""

import os
import sys
import traceback
from datetime import datetime

import pymysql
import psycopg2


# =========================================================
# CONFIGURACION MYSQL
# =========================================================

MYSQL_CONFIG = {
    "host": "localhost",
    "port": 3306,
    "user": "root",
    "password": "ELguru47",
    "database": "softcontrolweb",
    "charset": "utf8mb4"
}


# =========================================================
# CONFIGURACION POSTGRESQL
# =========================================================

POSTGRES_CONFIG = {
    "host": "201.232.132.51",
    "port": 5432,
    "user": "crm_user",
    "password": "crm_password",
    "dbname": "db_crm"
}


# =========================================================
# QUERY MYSQL
# =========================================================

MYSQL_QUERY = """
SELECT
    e.nombre,
    t.fecha_entrada,
    t.fecha_salida
FROM turnos_laborados t
INNER JOIN empleados e
    ON e.id = t.empleado_id
"""


# =========================================================
# SQL UPSERT POSTGRESQL
# =========================================================

POSTGRES_UPSERT = """
INSERT INTO turnos_laborados (
    nombre,
    fecha_entrada,
    fecha_salida
)
VALUES (%s, %s, %s)

ON CONFLICT (nombre, fecha_entrada)
DO UPDATE SET
    fecha_salida = EXCLUDED.fecha_salida
"""


# =========================================================
# CREAR TABLA SI NO EXISTE
# =========================================================

CREATE_TABLE_SQL = """
CREATE TABLE IF NOT EXISTS turnos_laborados (
    nombre TEXT NOT NULL,
    fecha_entrada TIMESTAMP NOT NULL,
    fecha_salida TIMESTAMP,
    fecha_sync TIMESTAMP DEFAULT NOW(),

    CONSTRAINT uq_turno
    UNIQUE(nombre, fecha_entrada)
)
"""


# =========================================================
# LOGS
# =========================================================

BASE_DIR = os.path.dirname(os.path.abspath(__file__))
LOG_DIR = os.path.join(BASE_DIR, "logs")

if not os.path.exists(LOG_DIR):
    os.makedirs(LOG_DIR)

LOG_FILE = os.path.join(
    LOG_DIR,
    f"sync_{datetime.now().strftime('%Y-%m-%d')}.txt"
)


def log(message):
    """
    Escribe mensajes en consola y archivo log.
    """

    timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")

    line = f"[{timestamp}] {message}"

    print(line)

    with open(LOG_FILE, "a", encoding="utf-8") as f:
        f.write(line + "\n")


# =========================================================
# MYSQL
# =========================================================

def connect_mysql():
    """
    Conexion MySQL.
    """

    return pymysql.connect(
        host=MYSQL_CONFIG["host"],
        port=MYSQL_CONFIG["port"],
        user=MYSQL_CONFIG["user"],
        password=MYSQL_CONFIG["password"],
        database=MYSQL_CONFIG["database"],
        charset=MYSQL_CONFIG["charset"],
        cursorclass=pymysql.cursors.DictCursor
    )


# =========================================================
# POSTGRESQL
# =========================================================

def connect_postgres():
    """
    Conexion PostgreSQL.
    """

    return psycopg2.connect(
        host=POSTGRES_CONFIG["host"],
        port=POSTGRES_CONFIG["port"],
        user=POSTGRES_CONFIG["user"],
        password=POSTGRES_CONFIG["password"],
        dbname=POSTGRES_CONFIG["dbname"]
    )


# =========================================================
# CREAR TABLA
# =========================================================

def create_table_if_not_exists(pg_conn):

    cursor = pg_conn.cursor()

    cursor.execute(CREATE_TABLE_SQL)

    pg_conn.commit()

    cursor.close()

    log("[OK] Tabla PostgreSQL verificada")


# =========================================================
# OBTENER DATOS MYSQL
# =========================================================

def fetch_mysql_data(mysql_conn):

    cursor = mysql_conn.cursor()

    cursor.execute(MYSQL_QUERY)

    rows = cursor.fetchall()

    cursor.close()

    log(f"[OK] Registros obtenidos MySQL: {len(rows)}")

    return rows


# =========================================================
# INSERTAR EN POSTGRESQL
# =========================================================

def sync_to_postgres(pg_conn, rows):

    cursor = pg_conn.cursor()

    total = 0

    for row in rows:

        try:

            nombre = row["nombre"]
            fecha_entrada = row["fecha_entrada"]
            fecha_salida = row["fecha_salida"]

            cursor.execute(
                POSTGRES_UPSERT,
                (
                    nombre,
                    fecha_entrada,
                    fecha_salida
                )
            )

            total += 1

        except Exception as e:

            log(f"[ERROR] Registro fallido: {str(e)}")

    pg_conn.commit()

    cursor.close()

    log(f"[OK] Registros sincronizados PostgreSQL: {total}")


# =========================================================
# MAIN
# =========================================================

def main():

    mysql_conn = None
    pg_conn = None

    try:

        log("================================================")
        log("INICIANDO SINCRONIZACION")
        log("================================================")

        # -------------------------------------------------
        # MYSQL
        # -------------------------------------------------

        log("[INFO] Conectando MySQL...")

        mysql_conn = connect_mysql()

        log("[OK] Conexion MySQL exitosa")

        # -------------------------------------------------
        # POSTGRESQL
        # -------------------------------------------------

        log("[INFO] Conectando PostgreSQL...")

        pg_conn = connect_postgres()

        log("[OK] Conexion PostgreSQL exitosa")

        # -------------------------------------------------
        # CREAR TABLA
        # -------------------------------------------------

        create_table_if_not_exists(pg_conn)

        # -------------------------------------------------
        # OBTENER DATOS
        # -------------------------------------------------

        rows = fetch_mysql_data(mysql_conn)

        if len(rows) == 0:

            log("[INFO] No existen registros para sincronizar")

            return

        # -------------------------------------------------
        # SYNC
        # -------------------------------------------------

        sync_to_postgres(pg_conn, rows)

        log("[OK] SINCRONIZACION FINALIZADA")

    except Exception as e:

        log("[FATAL ERROR]")
        log(str(e))

        traceback_str = traceback.format_exc()

        log(traceback_str)

    finally:

        try:

            if mysql_conn:
                mysql_conn.close()
                log("[OK] Conexion MySQL cerrada")

        except:
            pass

        try:

            if pg_conn:
                pg_conn.close()
                log("[OK] Conexion PostgreSQL cerrada")

        except:
            pass

        log("================================================")
        log("FIN DEL PROCESO")
        log("================================================")


# =========================================================
# ENTRYPOINT
# =========================================================

if __name__ == "__main__":

    main()
